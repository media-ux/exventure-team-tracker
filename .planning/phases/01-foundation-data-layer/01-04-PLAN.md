    ```typescript
    const addTeamMember = async (name: string, role: string, avatarUrl?: string) => {
      const { data, error } = await supabase
        .from('team_members')
        .insert({
          name,
          role,
          avatar_url: avatarUrl,
          company_id: companyId, // Hardcode for single-company MVP
          user_id: supabase.auth.getUser().data.user?.id // Link to auth user if applicable
        })
        .select()
        .single()

      if (error) throw error
      return data
    }
    ```

    Get tasks for team member:
    ```typescript
    const getTeamMemberTasks = async (memberId: string) => {
      const { data, error } = await supabase
        .from('tasks')
        .select(`
          id,
          title,
          status,
          due_date,
          sub_units(name, projects(name))
        `)
        .eq('assigned_to', memberId)

      if (error) throw error
      return data
    }
    ```
